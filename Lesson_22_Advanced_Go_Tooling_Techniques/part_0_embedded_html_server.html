package main import ( "embed" "fmt" "net/http" ) //go:embed assets/index.html var content embed.FS func main() { http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) { data, err := content.ReadFile("assets/index.html") if err != nil { http.Error(w, "File not found", http.StatusNotFound) return } w.Header().Set("Content-Type", "text/html") w.Write(data) }) fmt.Println("Serving on :8080") if err := http.ListenAndServe(":8080", nil); err != nil { fmt.Println(err) } }
